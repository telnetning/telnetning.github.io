Josephus 问题

分别使用数组和链表解决 Josephus 问题。
这是大概一年前写的一段程序了，自己看了好久才看明白过来，如果没有前面的思路，我觉得自己应该是看不懂了。这说明：

  • 程序写好注释是很重要的
  • 人真的很难了解当初的自己是怎么想的
  • 人的知识体系是变化着的

问题重述：设有 n 个人围坐在一个圆桌周围，现从第 s 个人开始报数，数到第 m 个人出列，然后从下一个人开始报数，数到第 m 个人再出列……求出列次序。
一年前的思路是：使用 n+1 长度的数组，0 下表的不使用（为了计算方便），每数到一个人，将它的至移到最后面，然后将这个下表后面所有未移动的数向前移动一位，直到剩下一个未移动的数。

程序代码如下：

#include <stdio.h>
#include <malloc.h>

int main(int argc, char **argv)
{
    int        n, m, s;   // n: the whole people
                          // m: the length to select a people
                          // s: begin from 

    int        s1;        // s1:current point people
    int        *p;        // p: point to array
    int        i, j;
    int        tmp;
    printf("Input n, m and s:\n");
    scanf("%d %d %d", &n, &m, &s);
    p = (int *)malloc((n + 1) * sizeof(int));
    
    //init
    s1 = s;
    for (i = 1;i <= n; i++){
        p[i] = i;
    }

    for (i = n; i >= 1; i--){
        s1 = (s1 + m - 1) % i;
        if (0 == s1){
            s1 = i;
        }
        tmp = p[s1];
        for (j = s1; j <= i - 1; j++ ){
            p[j] = p[j+1];
        }
        p[i] = tmp;
    }
    
    for (i = 1; i <= n / 2; i++){
        tmp = p[i];
        p[i] = p[n-i+1];
        p[n-i+1] = tmp;
    }

    //print the result
    printf("The people order is:\n");
    for(i = 1;i <= n;i++){
        printf("%d\t",p[i]);
    }
    printf("\n");

    return 0;
}

这种算法其中有大量的移动运算，导致程序效率不高，虽然空间复杂度较小。可以试着用链表重新实现，提高程序效率。
昨天用链表实现了这个问题。

/*use chain to solve Josephus problem
 * use cycle chain
*/
#include <stdio.h>
#include <stdlib.h>

struct Node;
typedef struct Node *PtrNode;
typedef PtrNode Position;

struct Node 
{
    int Element;
    Position Next;
};

int main(int argc, char **argv)
{
    int n, m, s;  //n:the whole people
                  //m:the length to select a people
                  //s:begin from

    int i;
    PtrNode newNode;
    PositioncurrentPosition;
    Position head, cursor;
    printf("input n(whole people), m(lenght to select) and s(begin from):");
    if(scanf("%d %d %d", &n, &m, &s)!=3){
        printf("scanf error!");
    }

    //Init
    for(i=1; i<=n; i++){
        newNode=malloc(sizeof(struct Node));
        newNode->Element=i;
        if(i==1){
            currentPosition=newNode; 
            head=newNode;
        } else {
            currentPosition->Next=newNode;
            currentPosition=newNode;
        }
    } 
    
    cursor=head;
    while(cursor->Next!=NULL){
        cursor=cursor->Next;
    }
    
    //point the last to the head,get a cycle chain
    cursor->Next=head;

    //start from s,cursor point to s-1
    if(s!=1){
        for(i=1;i<s;i++){
            cursor=cursor->Next;    
        }
    }

    //start the game
    if(m==1){
        cursor=cursor->Next;
        for(i=n;i>=1;i--){
            printf("%d ", cursor->Element);
            cursor=cursor->Next;
        }
        
    } else {
        while(cursor->Next!=cursor){
            for(i=1;i<m;i++){
                cursor=cursor->Next;
            }
            printf("%d ",cursor->Next->Element);
            cursor->Next=cursor->Next->Next;
        }

        printf("%d", cursor->Element);
    }
    return 0;
}
